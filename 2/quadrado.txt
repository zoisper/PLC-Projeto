main
{
  float[4] lados;
  int i;
  int j;
  int r;
  i = 4;
  
  while(i>0)
  {
    print("Digite lado: ");
    scan(lados[4 - i]);
    i = i - 1;  
  }
  
  

    r= 1;
    for(i=0;i<4;i=i+1)
    {
      for(j=i+1;j<4;j=j+1)
      {
        if(lados[i] != lados[j])
        {
          r = 0;
        }
      }
    }

    if(r and lados[0])
    {
      println("Podem ser os lados de um quadrado!");
    }
    else
    {
      println("NÃ£o podem ser os lados de um quadrado!");
    }

}
